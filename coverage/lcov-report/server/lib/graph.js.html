import Matrix from './matrix';
 
class Graph {
  constructor ({ size, incidentMatrix }) {
    this._size = size;
    this._vertices = {};
 
    if (incidentMatrix && incidentMatrix._size !== size) {
      throw new Error(`Graph: constructor: invalid size of incident matrix: ${incidentMatrix._size}. Expected: ${size}`);
    }
 
    for (let i = 0; i < size; i++) {
      if (incidentMatrix) {
        this._vertices[i] = incidentMatrix.getRowCoverage(i);
      } else {
        this._vertices[i] = [];
      }
    }
  }
 
  getIncidentMatrix (verticesList) {
    let size = this._size;
    if (verticesList && Array.isArray(verticesList)) {
      size = verticesList.length;
    }
    const resultMatrix = new Matrix(size);
 
    if (verticesList) {
      verticesList.forEach((origin, originIndex) => {
        this._vertices.forEach((target) => {
          const targetIndex = verticesList.indexOf(target);
          if (targetIndex !== -1) {
            resultMatrix.set(originIndex, targetIndex, 1);
          }
        });
      });
      return resultMatrix;
    }
    this._vertices.forEach((vertix, origin) => resultMatrix.set(origin, vertix, 1));
    return resultMatrix;
  }
 
  addEdge (origin, target) {
    if (!this._vertices[origin] || !this._vertices[target]) {
      throw new Error(`Graph: addEdge: Invalid arguments: (${origin})-->(${target})`);
    }
    if (!this._vertices[origin].includes(target)) {
      this._vertices[origin].push(target);
      return true;
    }
    return false;
  }
 
  removeEdge (origin, target) {
    if (!this._vertices[origin]) {
      throw new Error(`Graph: removeEdge: Invalid origin index: ${origin}`);
    }
    const targetIndex = this._vertices[origin].indexOf(target);
    if (targetIndex === -1) {
      return false;
    }
    delete this._vertices[origin][targetIndex];
    return true;
  }
 
  getSubgraph (verticesList) {
    if (verticesList.some((item) => !this._vertices[item])) {
      throw new Error(`Graph: getSubgraph: Invalid vertices list: ${verticesList}`);
    }
  }
}
 
export default Graph;
